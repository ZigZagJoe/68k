    move.l %d2,-(%sp)

    move.b (TDDR), %d0

	sub.w %d0,%d1
	move.w %d1,%d0

	| %d0 = (129 - TDDR) * 7 
	divu #129, %d0
	
	swap %d0
	clr.w %d0
	swap %d0
	| %d0 = ((129 - TDDR) * 7) / 129
	
	move.l %d2,%d1
	lsl.l #3,%d1
	sub.l %d2,%d1
	| %d1 = %d2 * 7
	
	add.l %d1, %d0
	
	move.l (millis_old), %d1
	cmp.l %d0, %d1
	jhi valid
	
	move.l %d1, %d0
	
valid:
	move.l %d0, (millis_old)
	
	move.l (%sp)+,%d2
    rts
    
millis_old: .space 4

#include <md5.h>

void task_md5() {
	printf("MD5 TASK (ID %d) started.\n", CURRENT_TASK_ID);
	uint8_t * ptr = 0x8000;
	
	yield();
	while(true) {
		while (!serial_available()) {
			if (millis() > 60000) {
				md5_state_t state;
				uint8_t digest[16];
	
				md5_init(&state);
				uint8_t *ptr2 = 0x8000;
				
				while (ptr != ptr2) {
					md5_append(&state, ptr2, 1);
					ptr2++;
				}
				
				md5_finish(&state, digest);

				char *hex_chars = "0123456789ABCDEF";
	
				for (uint8_t i = 0; i < 16; i++) {
					putc(hex_chars[digest[i]>>4]);
					putc(hex_chars[digest[i]&0xF]);
				}
	
				putc('\n');
				return;
			} else yield();
		}
		
		while (serial_available())
			*ptr++ = getc();
			
	}
}

void mem_dump(uint8_t *addr, uint32_t cnt) {
    int c = 0;
    char ascii[17];
    ascii[16] = 0;
    
    printf("\n%8X   ", addr);
    for (uint32_t i = 0; i < cnt; i++) {
        uint8_t b = *addr;
        
        ascii[c] = (b > 31 & b < 127) ? b : '.';
        
        printf("%02X ",b);
        
        addr++;
        
        if (++c == 16) {
            printf("  %s\n",ascii);
            if ((i+1) < cnt)
                printf("%8X   ", addr);
            c = 0;
        }
    }
    
    if (c && c < 15) {
        ascii[c] = 0; // null terminate
        while (c++ < 16)
            printf("   ");
        printf("  %s\n",ascii);
    } else printf("\n");
}

uint32_t hash( uint32_t a) {
    a = (a ^ 61) ^ (a >> 16);
    a = a + (a << 3);
    a = a ^ (a >> 4);
    a = a * 0x27d4eb2d;
    a = a ^ (a >> 15);
    return a;
}


uint32_t crc_region (uint8_t * start, uint32_t sz) {
	uint32_t crc = 0xDEADC0DE;
	
	for (uint32_t i = 0; i < sz; i++)
		crc = qcrc_update(crc, start[i]);
		
	printf("QCRC of %d bytes from 0x%x: 0x%X\n", sz,start, crc);
	return crc;
}
